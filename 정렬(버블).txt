//🧾 문제 : 2750번 - 수 정렬하기 (브론즈 2)
//문제
//N개의 수가 주어졌을 때, 이를 오름차순으로 정렬하는 프로그램을 작성하시오.
//
//입력
//첫째 줄에 수의 개수 N(1 ≤ N ≤ 1, 000)이 주어진다.
//둘째 줄부터 N개의 줄에는 수가 주어진다.
//이 수는 중복되지 않으며, 절댓값은 1, 000보다 작거나 같은 정수이다.
//
//출력
//첫째 줄부터 N개의 줄에 오름차순으로 정렬한 결과를 한 줄에 하나씩 출력한다.
//
//예제 입력 1
//5
//5
//2
//3
//4
//1
// 
//예제 출력 1
//1
//2
//3
//4
//5
//💡 핵심 아이디어
//단순한 정렬 알고리즘 문제입니다.
//
//C++에서는 sort() 함수를 이용하거나, 기본적인 버블 정렬 / 선택 정렬 / 삽입 정렬 구현으로도 해결할 수 있습니다.
//
//시간 제한이 2초이고 N이 최대 1, 000이므로, O(N²) 정렬도 통과됩니다.

#include <iostream>
#include <vector>
using namespace std;

int main() {
	int n;
	cin >> n;

	vector<int> v(n);

	for (int i = 0; i < n; i++) {
		cin >> v[i];
	}

	for (int i = 0; i < n; i++) {
		for (int j = 0; j < n; j++) {
			if (v[i] < v[j]) {
				int tmp = v[i];
				v[i] = v[j];
				v[j] = tmp;
			}
		}
	}

	for (int i = 0; i < n; i++) {
		cout << v[i] << "\n";
	}

}
