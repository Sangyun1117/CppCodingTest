#include <iostream>
using namespace std;


int main()
{
	ios::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);

	int n;

	cin >> n;

	int left = 0;
	int right = 0;
	int sum = 0;
	int count = 1;
	while (right != n) {
		if (sum == n) {
			count++;
			right++;
			sum += right;
		}
		else if (sum > n) {
			left++;
			sum -= left;
		}
		else {
			right++;
			sum += right;
		}
	}

	cout << count << "\n";
}
